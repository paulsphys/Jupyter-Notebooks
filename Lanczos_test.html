def W(x1,x2,L):
    lam = 2
    coeff = lam*(lam - 1)*(np.pi)**2/(2*L**2)
    Pot1 = coeff*(1.0/np.sin(np.pi*x1/L))**2
    #Hardwall1[abs(x1) >= R] = coff
    #Hardwall1[abs(x1) == R] = coff/2
    Pot2 = coeff*(1.0/np.sin(np.pi*x2/L))**2
    #Hardwall2[abs(x2) >= R] = coff
    #Hardwall2[abs(x2) == R] = coff/2
    
    return (Pot1 + Pot2)/hbar2by2m
